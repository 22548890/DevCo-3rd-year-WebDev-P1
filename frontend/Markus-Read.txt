let data = '';
const requestOpt = {
    method: 'GET',
    headers: {'Content-Type': 'application/json'},
    }
    async function fetchFunc() {
        return await fetch('http://127.0.0.1:5000/getContracts', requestOpt)
        .then(response => response.json())
        .catch(error => console.log(error));
    }
    (async () => {
        data = await fetchFunc();
    })()
